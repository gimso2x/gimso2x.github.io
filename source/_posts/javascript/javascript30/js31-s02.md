#### CSS + JS Clock

1. css 추가
```css
  /* css transform: rotate() 각도에 따라 회전 */
  transform: rotate(330deg);
  /* transform 기준점 0% ~ 100% 선 기준으로 0 왼쪽 끝, 100% 오른쪽 끝 */
  transform-origin: 100%;
  /* css 속성 변경할때 애니메이션 속도 조정 */
  transition: all 0.05s;
  /* 전환효과의 속도 곡선 지정, 여기선 시계바늘 반동역할 */
  transition-timing-function: cubic-bezier(0.1, 2.7, 0.58, 1);
```

2. 시계바늘 function
```javascript
  const secondHand = document.querySelector('.second-hand'); //secondHand 변수 지정
  const minHand = document.querySelector('.min-hand'); //minHand 변수 지정
  const hourHand = document.querySelector('.hour-hand'); //hourHand 변수 지정

  function setDate() {
    const now = new Date(); //now 변수에 Date 객체 지정(현재 시간, 날짜)

    const seconds = now.getSeconds(); //now에서 현재 초를 설정
    //now에서 현재 초에서 최대 초인 60을 나눈후 360을 곱한후
    //기본바늘이 각도가 -90도인 상태이므로 90도를 더함 이하 분, 시간 동일
    const secondsDegrees = ((seconds / 60) * 360) + 90;
    secondHand.style.transform = `rotate(${secondsDegrees}deg)`;

    const mins = now.getMinutes();
    const minsDegrees = ((mins / 60) * 360) + 90;
    minHand.style.transform = `rotate(${minsDegrees}deg)`;

    const hours = now.getHours();
    const hoursDegrees = ((hours / 60) * 360) + 90;
    hourHand.style.transform = `rotate(${hoursDegrees}deg)`;

  }
  //페이지 초기로드시 function 작동
  setDate();
  //1초마다 setDate function 작동으로 시계바늘 이동
  setInterval(setDate, 1000);
```
