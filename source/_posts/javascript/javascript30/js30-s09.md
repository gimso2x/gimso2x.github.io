#### Dev Tools Domination

1. script
```javascript
// Regular
//console.log 기본
console.log('hello');

// Interpolated
//console.log 변수로 값 넣음
console.log('hello i am a %s string!', '!!dd!!');

// Styled
//console.log 스타일 지정
console.log('%c I am some greate text', 'font-size:50px;background:red;');

// warning!
//console warning
console.warn('OH NOOO');

// Error :|
//console error
console.error('Shit');

// Info
console.info("Cr eat 3-4 people");

// Testing
// 평가된 식이 false이면 콘솔에 오류를 기록합니다.
const p = document.querySelector('p');
console.assert(p.classList.contains('ouch'), 'That is wrong!');

// clearing
// 콘솔을 지웁니다.
console.clear();

// Viewing DOM Elements
console.log(p);
// 지정된 객체의 자바스크립트 표현을 출력합니다. 로그 중인 객체가 HTML 요소인 경우 DOM 표현의 속성이 아래와 같이 출력됩니다.
console.dir(p);

console.clear();

// Grouping together
// foreach 문안에서 그룹 묶음 ${dog.name}이란 이름 지정
dogs.forEach(dog => {
	console.group(`${dog.name}`);
	console.log(`This is ${dog.name}`);
	console.log(`${dog.name} is ${dog.age} years old`);
	console.log(`${dog.name} is ${dog.age * 7} dog years old`);
	console.groupEnd(`${dog.name}`);
})

// counting
// count()가 호출된 횟수를 동일한 줄에 동일한 레이블을 사용하여 기록합니다.
console.count('Wes');
console.count('Wes');
console.count('Steve');
console.count('Steve');
console.count('Wes');
console.count('Steve');
console.count('Wes');
console.count('Steve');
console.count('Steve');
console.count('Steve');
console.count('Steve');
console.count('Steve');

// timing
// timeend가 날때까지 시간 기록
console.time('fetching data');
fetch('https://api.github.com/users/wesbos')
	.then(data => data.json())
	.then(data => {
	console.timeEnd('fetching data');
	console.log(data);
	})
```
