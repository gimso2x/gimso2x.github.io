#### Fun with HTML5 Canvas

1. script
```javascript
const canvas = document.querySelector("#draw");
// getContext 메소드를 사용하여 canvas 렌더링 컨텍스트와 그리기 함수들 사용
const ctx = canvas.getContext("2d");
// canvas width, height를 window 사이즈로 지정
canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

//ctx - canvas 2d에서 strokeStyle 색을 #BADA55로 지정
ctx.strokeStyle = "#BADA55";
//둥근 모서리
ctx.lineJoin = "round";
//둥근 끝이 있는 선
ctx.lineCap = "round";
//선 두께 100
ctx.lineWidth = 100;

let isDrawing = false;
let lastX = 0;
let lastY = 0;
let hue = 0;
let direction = true;

function draw(e) {
if (!isDrawing) return;
console.log(e);
//hsl(색상, 채도, 명도)
ctx.strokeStyle = `hsl(${hue}, 100%, 50%)`;
//canvas 메소드 선그리기할때
ctx.beginPath();
//mouse 눌럿을때 지정된 변수값으로 canvas 좌표 이동
ctx.moveTo(lastX, lastY);
//moveTo에서 시작해서 LineTo 좌표까지 지정
ctx.lineTo(e.offsetX, e.offsetY);
//stroke로 라인 생성
ctx.stroke();
//마지막 좌표로 lastX, lastY 값 지정
[lastX, lastY] = [e.offsetX, e.offsetY];

//함수 작동될때마다 hue값 1씩 증가, 360이 넘어가면 0으로 초기화
hue++;
if (hue >= 360) {
	hue = 0;
}

// canvas 요소의 라인 두께가 100이상이거나 1이하이면 dircetion 함수 false처리
if (ctx.lineWidth >= 100 || ctx.lineWidth <= 1) {
	direction = !direction;
}

// canvas요소 linewidth 증감, 증가 - dircetion 상태에 따라
if (direction) {
	ctx.lineWidth++;
} else {
	ctx.lineWidth--;
}
}

//canvas에 mousedown 이벤트 감지될 때 해당 좌표값 변수 값 지정
canvas.addEventListener("mousedown", e => {
isDrawing = true;
[lastX, lastY] = [e.offsetX, e.offsetY];
});
//canvas에 mousemove 이벤트 감지될 때 위에서 지정된 draw 함수 동작
canvas.addEventListener("mousemove", draw);
//canvas에 마우스 버튼을 눌렀다 뗄때 동작
canvas.addEventListener("mouseup", () => (isDrawing = false));
//canvas에 마우스 떠날 때 동작
canvas.addEventListener("mouseout", () => (isDrawing = false));
```
