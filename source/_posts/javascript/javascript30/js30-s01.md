2. 템플릿 리터럴은 내장된 표현식을 허용하는 문자열 리터럴입니다. 여러 줄로 이뤄진 문자열과 문자 보간기능을 사용할 수 있습니다. 이전 버전의 ES2015사양 명세에서는 "template strings" (템플릿 문자열) 라고 불려 왔습니다.
템플릿 리터럴은 이중 따옴표 나 작은 따옴표 대신 백틱(` `)을 이용합니다.
```javascript
var a = 5;
var b = 10;
console.log(`Fifteen is ${a + b} and
not ${2 * a + b}.`);
// "Fifteen is 15 and
// not 20."
```
3. 재생 동작 function
```javascript
  function playSound(e) {
    // audio 태그중에 속성이 data-key가 keydown 이벤트가 발생된 값을 가진 태그를 담는다
    const audio = document.querySelector(`audio[data-key="${e.keyCode}"`);
    // class가 key를 가진 태그중에 속성이 data-key가 keydown 이벤트가 발생된 값을 가진 태그를 담는다
    const key = document.querySelector(`.key[data-key="${e.keyCode}"`);

    //audio태그가 비었을 때 함수동작 멈춤
    if(!audio) return;
    //audio태그 파일 재생지점 0 셋팅
    audio.currentTime = 0;
    //audio태그 재생
    audio.play();

    //key태그에 playing class 추가
    key.classList.add('playing');
  }
```

4. transition 끝난후 class 제거 function
```javascript
  function removeTransition(e) {
    //이 function안에 들어왔을때 element요소의 propertyName이 transform 이 아닐경우 함수동작 멈춤
    if (e.propertyName !== 'transform') return;
    //this = keys , keys에 담긴 태그들에에서 playing class 제거
    this.classList.remove('playing');
  }
```

5. 기본 동작
```javascript
  //keys 라는 변수에 key라는 class를 가진 모든 태그를 담는다
  const keys = document.querySelectorAll('.key');
  //keys 변수 forEach문을 돌려 'transitionend' 동작이 실행되었을때 removeTransition function 을 실행
  keys.forEach(key => key.addEventListener('transitionend', removeTransition));
  //'keydown' 이벤트가 실행되었을 때 playSound function을 실행
  window.addEventListener('keydown', playSound);
```
