#### Array Cardio Day 1

1. filter
```javascript
// filter - 주어진 조건을 통과하는 요소를 모아 새로운 배열로 반환(1500년대 출생)
const fiften = inventors.filter(function(inventor) {
  if(inventor.year > 1500 && inventor.year < 1600) {
    return true;
  }
});
console.log(fiften);
```

2. map
```javascript
// map - 배열내의 모든요소 각각에 주어진 함수를 호출한 결과를 모아 새로운 배열 반환(first name과 last name 붙여서 호출)
const FULLNAME = inventors.map(inventor => `${inventor.first} ${inventor.last}`)
console.log(FULLNAME);
```

3. sort
```javascript
// sort - 배열내에서 조건식에 따라 비교하고 0보다 작을경우와 0보다 클경우로 오는 순서 재배열
const ORDERED = inventors.sort((a, b) => a.year > b.year ? 1 : -1);
console.table(ORDERED);
```

4. reduce
```javascript
// reduce - 배열.reduce((누적값, 현잿값, 인덱스, 요소) => { return 결과 }, 초깃값);
const TOTALYERS = inventors.reduce((total, inventor) => {
  return total +  (inventor.passed - inventor.year);
}, 0);
console.log(TOTALYERS);
```
